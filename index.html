    function scrollToId(id){
      const el = document.getElementById(id);
      if(el) el.scrollIntoView({behavior:"smooth", block:"start"});
    }

    function toast(msg){
      const t = document.getElementById("toast");
      t.textContent = msg;
      t.classList.add("show");
      setTimeout(()=>t.classList.remove("show"), 2400);
    }

    function loadJSON(key, fallback){
      try{
        const raw = localStorage.getItem(key);
        return raw ? JSON.parse(raw) : fallback;
      }catch(e){ return fallback; }
    }
    function saveJSON(key, value){
      localStorage.setItem(key, JSON.stringify(value));
    }

    // Profile
    function applyProfile(){
      const name = document.getElementById("gamerName").value.trim() || "MaddenGamer";
      const console = document.getElementById("console").value;
      const mode = document.getElementById("mode").value;

      const profile = {name, console, mode};
      saveJSON(LS_KEYS.profile, profile);

      document.getElementById("footerName").textContent = name;
      document.title = name + "  Madden Gamer Hub";

      // Fun stats
      document.getElementById("modeStat").textContent = mode + "  " + console;
      document.getElementById("styleStat").textContent = "Balanced + Reads";
      document.getElementById("schemeStat").textContent = "Spread / West Coast";
      document.getElementById("goalStat").textContent = "Weekly wins + clips";

      toast("Profile applied ");
    }

    // Plays
    let plays = loadJSON(LS_KEYS.plays, []);
    function addPlay(){
      const formation = document.getElementById("formation").value.trim();
      const playName = document.getElementById("playName").value.trim();
      const playType = document.getElementById("playType").value;
      const situation = document.getElementById("situation").value.trim();
      const notes = document.getElementById("notes").value.trim();

      if(!formation || !playName){
        toast("Add formation + play name first.");
        return;
      }

      plays.unshift({
        id: crypto.randomUUID(),
        formation, playName, playType, situation, notes,
        createdAt: new Date().toISOString()
      });

      saveJSON(LS_KEYS.plays, plays);
      renderPlays();

      document.getElementById("formation").value = "";
      document.getElementById("playName").value = "";
      document.getElementById("situation").value = "";
      document.getElementById("notes").value = "";
      toast("Play added ");
    }

    function removePlay(id){
      plays = plays.filter(p=>p.id !== id);
      saveJSON(LS_KEYS.plays, plays);
      renderPlays();
      toast("Play removed.");
    }

    function renderPlays(){
      const tbody = document.getElementById("playsTbody");
      tbody.innerHTML = "";
      plays.forEach(p=>{
        const tr = document.createElement("tr");
        tr.innerHTML = `
          <td>${escapeHTML(p.formation)}</td>
          <td>
